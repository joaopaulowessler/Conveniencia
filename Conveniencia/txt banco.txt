CREATE TABLE CLIENTE
(CLI_CODIGO INTEGER NOT NULL,
 CLI_NOME VARCHAR(50),
 CLI_CPF VARCHAR(20),
 CLI_RG VARCHAR(20),
 CLI_SEXO VARCHAR(15),
 CLI_TELEFONE VARCHAR(20),
 CLI_EMAIL VARCHAR(50),
 CLI_CIDADE VARCHAR(50),
 CLI_ESTADO VARCHAR(5),
 CLI_ENDERECO VARCHAR(50),
 PRIMARY KEY (CLI_CODIGO));

CREATE TABLE FUNCIONARIO
(FUN_CODIGO INTEGER NOT NULL,
 FUN_NOME VARCHAR(50),
 FUN_CPF VARCHAR(20),
 FUN_RG VARCHAR(20),
 FUN_SEXO VARCHAR(2),
 FUN_TELEFONE VARCHAR(20),
 FUN_EMAIL VARCHAR(50),
 FUN_CIDADE VARCHAR(50),
 FUN_ESTADO VARCHAR(5),
 FUN_ENDERECO VARCHAR(50),
 PRIMARY KEY (FUN_CODIGO));

CREATE TABLE PRODUTO
(PRO_CODIGO INTEGER NOT NULL,
 PRO_DESCRICAO VARCHAR(50),
 PRO_UNIDADE VARCHAR(20),
 PRO_FORNECEDOR VARCHAR(20),
 PRO_PRECO DECIMAL(10,2),
 PRO_DATACAD VARCHAR(15),
 PRIMARY KEY (PRO_CODIGO));

CREATE TABLE VENDA
(VEN_CODIGO INTEGER NOT NULL,
 CLI_CODIGO INTEGER,
 FUN_CODIGO INTEGER,
 PRO_CODIGO INTEGER,
 VEN_QUANTIDADE DECIMAL(10,2),
 VEN_OBSERVACAO VARCHAR(100),
 PRIMARY KEY (PRO_CODIGO));
